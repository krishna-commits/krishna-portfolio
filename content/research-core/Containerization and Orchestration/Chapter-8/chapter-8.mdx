### Skeleton
```bash
[root@sanjeeb ~]# cd /etc/skel 
[root@sanjeeb skel]# ls

# Viewing hidden files 
[root@sanjeeb skel]# ls -a 
.  ..  .bash_logout  .bash_profile  .bashrc  .mozilla

# Creating a new file 
[root@sanjeeb skel]# vim user_guildelines 
[root@sanjeeb skel]# cat user_guildelines 
User Guildelines : 
1. User should poweroff the system after use. 
2. User should not try to break the root password. 
3. User should not use the server more than 1 hour.

[root@sanjeeb skel]# ls 
user_guildelines

# Creating a new user to check if user_guildelines is added automatically in user creation
[root@sanjeeb skel]# useradd naken 
[root@sanjeeb skel]# passwd naken 
# Switching another user 
[root@sanjeeb skel]# su - naken 
[naken@sanjeeb ~]$ ls
user_guildelines

```
