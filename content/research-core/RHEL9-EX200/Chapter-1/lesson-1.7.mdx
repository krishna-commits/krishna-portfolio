```bash 
[sanjeeb@server folder]$ ls

[sanjeeb@server folder]$ touch file1
[sanjeeb@server folder]$ ls
file1

[sanjeeb@server folder]$ touch file{1,2,3}
[sanjeeb@server folder]$ ls
file1  file2  file3

[sanjeeb@server folder]$ touch file{1..5}
[sanjeeb@server folder]$ ls
file1  file2  file3  file4  file5

[sanjeeb@server folder]$ mkdir -p folder1/{file1,file2}
[sanjeeb@server folder]$ tree
.
├── file1
├── file2
├── file3
├── file4
├── file5
└── folder1
    ├── file1
    └── file2
```
